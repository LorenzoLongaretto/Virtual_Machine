m1 equ "palindr"
m2 equ "no palin"
MOV DX, 0
MOV CX, 50
MOV AX, %000
SYS %3

slen [bx], [0] ;apunta al final
sub [bx],1 ;por el \0
mov [dx],[0] ;apunta al principio


sigo: cmp [dx],[bx] ;comparo la posicion de cada reg, salto si el cmp es -
jnp fin

cmp bx,dx ;comparo si son =
jnz fin2
add dx,1
sub bx,1
jmp sigo

fin: mov dx, m1 ;son palind
mov cx 10
mov ax, 1
sys %4
stop

fin2: mov dx,m2 ;no son palind
mov cx 10
mov ax, 1
sys %4
stop
